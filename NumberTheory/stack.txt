class Stack {
	// Define the data members
    Node* head;
    int size;
   public:
    Stack() {
        // Implement the Constructor
		head=NULL;
        size=0;
    }

	/*----------------- Public Functions of Stack -----------------*/

    int getSize() {
        // Implement the getSize() function
        return size;
    }

    bool isEmpty() {
        // Implement the isEmpty() function
         return (head==NULL);
    }

    void push(int element) {
        // Implement the push() function
        Node* newNode = new Node(element);
        if(head==NULL){
            head=newNode;
            head->next=NULL;
        }
        else{
            newNode->next=head;
            head=newNode;
        }
        size++;
    }

    int pop() {
        // Implement the pop() function
        if(!head)return -1;
        int ans=head->data;
        Node* newhead=head->next;
        delete head;
        head=newhead;
        size--;
        return ans;
    }

    int top() {
        // Implement the top() function\
        if(!head)return -1;
        return head->data;
    }
};
